$$
F\lbrack a_{1}f_{1}(x)+a_{2}f_{2}(x)\rbrack=a_{1}F\lbrack f_{1}(x)\rbrack+a_{2}F\lbrack f_{2}(x)\rbrack &线性定理
\ F[f(x-x_{0})]=e^{-i k x_{0}}F[f(x)] &延迟定理
\F[f(x-x_{0})]=\int_{-\infty}^{\infty}f(x-x_{0})e^{-i k x}d x
\= \int_{\infty}^{\infty}!f(x-x_{0}){\color {red} e^{-i k(x-x_{0})}e^{-i k x_{0}}}d x
\=e^{-i k x_{0}}\int_{-\infty}^{\infty}!f(x)e^{-i k x}d x
\ F[f(x)e^{i k_{0}x}]=c(k-k_0),~~~c(k)\equiv F[f(x)] & 位移定理
\ F[f(x)e^{i k_{0}x}]=\int_{-\infty}^{\infty}f(x)e^{i k_{0}x}e^{-i k x}d x
\=\int_{-\infty}^{\infty}!f(x)e^{-i(k-k_{0})x}d x
\ F\left[f^{\prime}(x)\right]=i k F[f(x)], \quad F\left[f^{(n)}(x)\right]=(i k)^{n} F[f(x)] & 微分定理
\F\left[f^{\prime}(x)\right]=\int_{-\infty}^{\infty} f^{\prime}(x) e^{-i k x} d x
\=\left.f(x) e^{-i k x}\right|{-\infty} ^{\infty}+i k \int{-\infty}^{\infty} f(x) e^{-i k x} d x 【分部积分】
\F\left[f_{1}(x)\right] \cdot F\left[f_{2}(x)\right]=F\left[f_{1}(x) * f_{2}(x)\right] &卷积定理 \
f_{1}(x) * f_{2}(x)=\int_{-\infty}^{\infty} f_{1}(x-\xi) f_{2}(\xi) d \xi \
F\left[f_{1}(x) * f_{2}(x)\right]=\int_{-\infty}^{\infty}\left[\int_{-\infty}^{\infty} f_{1}(x-\xi) e^{-i k x} d x\right] f_{2}(\xi) d \xi
\=F\left[f_{1}(x)\right] \int_{-\infty}^{\infty} f_{2}(\xi) e^{-i k \xi} d \xi
\ F[f(a x)]=\frac{1}{|a|} c\left(\frac{k}{a}\right), \quad a \neq 0 &坐标缩放定理
$$
